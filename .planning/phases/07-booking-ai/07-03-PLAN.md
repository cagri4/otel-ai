**Add `summarizeOldTurns()` — internal function (not exported):**

```typescript
async function summarizeOldTurns(conversationId: string, turnsToSummarize: number): Promise<void> {
  // Use service client — this runs fire-and-forget, session may not be available
  const { createServiceClient } = await import('@/lib/supabase/service');
  const supabase = createServiceClient();

  // Check if summary is already up to date
  const { data: existing } = await supabase
    .from('conversation_summaries')
    .select('turns_summarized')
    .eq('conversation_id', conversationId)
    .maybeSingle();

  if (existing && existing.turns_summarized >= turnsToSummarize) {
    return; // Summary already covers these turns — skip (Pitfall 4)
  }

  // Load the older turns that need summarizing (all turns BEFORE the recent N)
  const { data: allTurns } = await supabase
    .from('conversation_turns')
    .select('role, content, created_at')
    .eq('conversation_id', conversationId)
    .order('created_at', { ascending: true })
    .limit(turnsToSummarize);

  if (!allTurns || allTurns.length === 0) return;

  // Format turns as readable text for summarization
  const turnText = allTurns
    .filter((t) => t.role === 'user' || t.role === 'assistant')
    .map((t) => `${t.role}: ${t.content}`)
    .join('\n');

  if (!turnText.trim()) return;

  // Call Claude sonnet for summarization (internal/background task = sonnet per project decision)
  const Anthropic = (await import('@anthropic-ai/sdk')).default;
  const client = new Anthropic();

  const response = await client.messages.create({
    model: 'claude-sonnet-4-6',
    max_tokens: 1024,
    system: 'You are a conversation summarizer. Produce a concise summary of the following hotel guest conversation. Focus on: guest preferences, dates discussed, room types considered, pricing mentioned, any special requests. Output only the summary, nothing else.',
    messages: [{ role: 'user', content: turnText }],
  });

  const summaryText = response.content
    .filter((block): block is { type: 'text'; text: string } => block.type === 'text')
    .map((block) => block.text)
    .join('');

  if (!summaryText.trim()) return;

  // Get hotel_id from the first turn (needed for RLS/FK)
  const { data: firstTurn } = await supabase
    .from('conversation_turns')
    .select('hotel_id')
    .eq('conversation_id', conversationId)
    .limit(1)
    .single();

  if (!firstTurn) return;

  // Upsert summary — service client has full access
  await supabase
    .from('conversation_summaries')
    .upsert({
      conversation_id: conversationId,
      hotel_id: firstTurn.hotel_id,
      summary: summaryText.trim(),
      turns_summarized: turnsToSummarize,
      updated_at: new Date().toISOString(),
    }, { onConflict: 'conversation_id' });
}
```

Key details:
- Uses `createServiceClient()` (not session client) because this runs fire-and-forget in async context where the session cookie may no longer be available.
- Checks `turns_summarized` before re-summarizing (Pitfall 4 prevention).
- Only includes user/assistant turns in the summary text (tool turns are implementation details).
- Uses `claude-sonnet-4-6` per project decision for internal/background tasks.
- Upserts into `conversation_summaries` with `onConflict: 'conversation_id'`.
