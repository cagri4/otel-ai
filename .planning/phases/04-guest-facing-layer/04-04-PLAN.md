2. Create `src/i18n/request.ts` (next-intl locale resolution without URL routing):
   ```typescript
   import { getRequestConfig } from 'next-intl/server';
   import { cookies } from 'next/headers';

   export default getRequestConfig(async () => {
     const cookieStore = await cookies();
     const locale = cookieStore.get('NEXT_LOCALE')?.value ?? 'en';
     const validLocales = ['en', 'tr'];
     const resolvedLocale = validLocales.includes(locale) ? locale : 'en';

     return {
       locale: resolvedLocale,
       messages: (await import(`../../messages/${resolvedLocale}.json`)).default,
     };
   });
   ```

6. Update `src/app/layout.tsx` — wrap with NextIntlClientProvider:
   ```typescript
   import { NextIntlClientProvider } from 'next-intl';
   import { getMessages, getLocale } from 'next-intl/server';

   export default async function RootLayout({ children }) {
     const locale = await getLocale();
     const messages = await getMessages();
     return (
       <html lang={locale}>
         <body>
           <NextIntlClientProvider messages={messages}>
             {children}
           </NextIntlClientProvider>
         </body>
       </html>
     );
   }
   ```
   Keep existing metadata export. Add `locale` to the `<html lang>` attribute.
